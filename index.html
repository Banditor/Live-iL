async function loadChannels() { try { // עדכן את ה-URL כדי להצביע על המיקום של ה-JSON const response = await fetch('URL_TO_YOUR_CHANNELS_JSON'); const data = await response.json(); const channels = data.TvChannels; const tableBody = document.querySelector('#channels-table tbody'); channels.forEach(channel => { const row = document.createElement('tr'); // תא של הלוגו const logoCell = document.createElement('td'); const logoImg = document.createElement('img'); logoImg.src = `path/to/images/${channel.image}`; // עדכן את הנתיב ללוגו logoImg.alt = `${channel.name} logo`; logoImg.classList.add('channel-logo'); logoCell.appendChild(logoImg); row.appendChild(logoCell); // תא של שם הערוץ const nameCell = document.createElement('td'); nameCell.textContent = channel.name; row.appendChild(nameCell); // תא של כפתור הצפייה const watchCell = document.createElement('td'); const watchButton = document.createElement('button'); watchButton.textContent = 'Watch'; watchButton.onclick = () => { // נדרש לעדכן עם כתובת הזרימה המתאימה window.open(`streaming_url_for_${channel.module}`, '_blank'); }; watchCell.appendChild(watchButton); row.appendChild(watchCell); tableBody.appendChild(row); }); } catch (error) { console.error('Error loading channels:', error); } }
